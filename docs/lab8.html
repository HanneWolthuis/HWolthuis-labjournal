library(RSiena)

mynet1 <- sienaDependent(array(c(s501, s502, s503), dim=c(50, 50, 3)))
mybeh <- sienaDependent(s50a, type="behavior")
smoke <- s50s
smoke <- varCovar(s50s)

drinking <- varCovar(s50a)

mydata <- sienaDataCreate(mynet1, mybeh, drinking, smoke)
print01Report(mydata)
myeff <- getEffects(mydata)
myeff <- includeEffects(myeff, unequalX, name = "mynet1", interaction1 = "smoke")
#>   effectNumber effectName    shortName include fix   test  initialValue parm
